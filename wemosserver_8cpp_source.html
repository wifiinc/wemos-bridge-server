225 exit(EXIT_FAILURE);
226 }
227
228 const int enable_opt = 1;
229 if (setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR | SO_REUSEPORT, &enable_opt,

231 perror("setsockopt() failed");
232 throw std::runtime_error("setsockopt() failed");
233 exit(EXIT_FAILURE);
234 }
235
236 if (bind(server_fd, (struct sockaddr *)&listen_address, sizeof(listen_address)) < 0) {
237 perror("bind() failed");
238 throw std::runtime_error("bind() failed");
239 exit(EXIT_FAILURE);
240 }
241
242 if (listen(server_fd, MAX_CLIENTS) < 0) {
243 perror("listen() failed");

245 exit(EXIT_FAILURE);
246 }

248 std::cout << "Listening on port " << ntohs(listen_address.sin_port) << " (max " << MAX_CLIENTS
